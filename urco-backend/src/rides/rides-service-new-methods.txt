/**
 * Démarrer un trajet (chauffeur)
 * - Mettre le trajet en état IN_PROGRESS
 * - Générer les codes de sécurité pour les passagers
 * - Créer un événement d'audit
 */
async startRide(rideId: string, driverId: string) {
  // Vérifier que le trajet existe et appartient au chauffeur
  const ride = await this.prisma.ride.findUnique({
    where: { id: rideId },
    include: { bookings: true },
  });

  if (!ride) {
    throw new NotFoundException('Trajet non trouvé');
  }

  if (ride.driverId !== driverId) {
    throw new ForbiddenException('Seul le chauffeur peut démarrer ce trajet');
  }

  if (ride.state !== 'PENDING') {
    throw new BadRequestException(`Le trajet n'est pas en état PENDING (état actuel: ${ride.state})`);
  }

  // Mettre à jour le trajet
  const updatedRide = await this.prisma.ride.update({
    where: { id: rideId },
    data: {
      state: 'IN_PROGRESS',
      actualStartTime: new Date(),
    },
    include: {
      driver: {
        select: {
          id: true,
          username: true,
          firstName: true,
          lastName: true,
          avatar: true,
        },
      },
      bookings: {
        include: {
          passenger: {
            select: {
              id: true,
              username: true,
              firstName: true,
              lastName: true,
            },
          },
        },
      },
    },
  });

  // Créer un événement d'audit
  await this.ridesEventsService.createRideEvent(
    rideId,
    'STARTED_BY_DRIVER',
    driverId,
    { startTime: new Date() },
    'Le chauffeur a démarré le trajet'
  );

  // Générer les codes de sécurité pour les réservations confirmées
  const confirmedBookings = ride.bookings.filter((b) => b.status === 'CONFIRMED');

  for (const booking of confirmedBookings) {
    const securityCode = Math.floor(1000 + Math.random() * 9000).toString();
    const codeExpiry = new Date(Date.now() + 30 * 60 * 1000); // Code valide pendant 30 minutes

    await this.prisma.booking.update({
      where: { id: booking.id },
      data: {
        securityCode,
        codeExpiry,
        status: 'CODE_SENT',
      },
    });

    // Créer un événement pour chaque code envoyé
    await this.ridesEventsService.createRideEvent(
      rideId,
      'PASSENGER_CODE_SENT',
      booking.passengerId,
      { bookingId: booking.id, code: securityCode, expiresAt: codeExpiry },
      `Code de sécurité envoyé au passager ${booking.passenger.firstName} ${booking.passenger.lastName}`
    );

    // TODO: Envoyer SMS/Push avec le code
    console.log(`✉️ [TODO] Envoyer code ${securityCode} à ${booking.passenger.username}`);
  }

  return updatedRide;
}

/**
 * Terminer un trajet (chauffeur)
 * - Marquer le trajet comme COMPLETED
 * - Marquer les réservations comme COMPLETED
 * - Créer un événement d'audit
 */
async completeRide(rideId: string, driverId: string) {
  const ride = await this.prisma.ride.findUnique({
    where: { id: rideId },
    include: { bookings: true },
  });

  if (!ride) {
    throw new NotFoundException('Trajet non trouvé');
  }

  if (ride.driverId !== driverId) {
    throw new ForbiddenException('Seul le chauffeur peut terminer ce trajet');
  }

  if (ride.state !== 'IN_PROGRESS') {
    throw new BadRequestException(`Le trajet n'est pas en cours (état actuel: ${ride.state})`);
  }

  // Mettre à jour le trajet
  const completedRide = await this.prisma.ride.update({
    where: { id: rideId },
    data: {
      state: 'COMPLETED',
      actualEndTime: new Date(),
    },
    include: {
      driver: true,
      bookings: {
        include: {
          passenger: {
            select: {
              id: true,
              username: true,
              firstName: true,
              lastName: true,
            },
          },
        },
      },
    },
  });

  // Mettre à jour les réservations
  await this.prisma.booking.updateMany({
    where: {
      rideId,
      status: { in: ['CODE_VERIFIED', 'RIDE_IN_PROGRESS'] },
    },
    data: {
      status: 'COMPLETED',
    },
  });

  // Créer un événement d'audit
  await this.ridesEventsService.createRideEvent(
    rideId,
    'RIDE_COMPLETED',
    driverId,
    { endTime: new Date() },
    'Le trajet a été complété'
  );

  return completedRide;
}

/**
 * Annuler un trajet (chauffeur ou admin)
 */
async cancelRide(rideId: string, userId: string, reason?: string) {
  const ride = await this.prisma.ride.findUnique({
    where: { id: rideId },
  });

  if (!ride) {
    throw new NotFoundException('Trajet non trouvé');
  }

  // Vérifier que c'est le chauffeur ou un admin
  const user = await this.prisma.user.findUnique({
    where: { id: userId },
  });

  if (ride.driverId !== userId && !user?.isAdmin) {
    throw new ForbiddenException('Vous ne pouvez pas annuler ce trajet');
  }

  if (ride.state === 'COMPLETED') {
    throw new BadRequestException('Impossible d\'annuler un trajet déjà complété');
  }

  // Annuler le trajet
  const cancelledRide = await this.prisma.ride.update({
    where: { id: rideId },
    data: {
      state: 'CANCELLED',
    },
  });

  // Annuler les réservations
  await this.prisma.booking.updateMany({
    where: {
      rideId,
      status: { not: 'COMPLETED' },
    },
    data: {
      status: 'CANCELLED',
    },
  });

  // Créer un événement d'audit
  await this.ridesEventsService.createRideEvent(
    rideId,
    'RIDE_CANCELLED',
    userId,
    { reason },
    `Le trajet a été annulé: ${reason || 'Raison non spécifiée'}`
  );

  return cancelledRide;
}

/**
 * Obtenir l'historique complet d'un trajet (avec audit trail)
 */
async getRideFullHistory(rideId: string) {
  const ride = await this.getRideById(rideId);
  // Assurer que RideEvent model existe
  const events = await this.prisma.rideEvent.findMany({
    where: { rideId },
    orderBy: { createdAt: 'asc' },
    include: {
      user: {
        select: {
          id: true,
          username: true,
          firstName: true,
          lastName: true,
        },
      },
    },
  });

  return {
    ride,
    events,
  };
}

/**
 * Obtenir les statistiques détaillées d'un trajet
 */
async getRideDetailedStats(rideId: string) {
  const ride = await this.getRideById(rideId);
  // Assurer que RideEvent et RideTracking models existent
  const events = await this.prisma.rideEvent.findMany({
    where: { rideId },
  });

  const trackings = await this.prisma.rideTracking.findMany({
    where: { rideId },
    orderBy: { timestamp: 'asc' },
  });

  const confirmedBookings = ride.bookings.filter((b) => b.status === 'COMPLETED');
  const totalSeatsBooked = confirmedBookings.reduce((sum, b) => sum + b.seats, 0);

  return {
    ride: {
      id: ride.id,
      state: ride.state,
      origin: ride.origin,
      destination: ride.destination,
      departureDate: ride.departureDate,
      actualStartTime: ride.actualStartTime,
      actualEndTime: ride.actualEndTime,
      pricePerSeat: ride.pricePerSeat,
      availableSeats: ride.availableSeats,
    },
    stats: {
      totalEvents: events.length,
      locationUpdates: trackings.length,
      totalPassengers: ride.bookings.length,
      confirmedPassengers: confirmedBookings.length,
      totalSeatsBooked,
      estimatedRevenue: ride.pricePerSeat * totalSeatsBooked,
    },
    timeline: {
      createdAt: ride.createdAt,
      startedAt: ride.actualStartTime,
      completedAt: ride.actualEndTime,
      duration: ride.actualStartTime && ride.actualEndTime 
        ? Math.floor((ride.actualEndTime.getTime() - ride.actualStartTime.getTime()) / 1000 / 60)
        : null,
    },
  };
}
